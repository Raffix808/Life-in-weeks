// ── CONSTANTS ────────────────────────────────

const LIFE_EXP  = 81;

const COLS      = 52;

const TOTAL_ROWS = LIFE_EXP;

const SLEEP_H   = 7.5;

const WORK_H    = 8;

const EAT_H     = 1.5;

const CHORES_H  = 1.0;

const HRS_WEEK  = 168;


const CATEGORIES = [

  { key:'sleep',   color:'#3B5BA5', label:'Sleeping',             emoji:'🌙', note:'7.5 hrs/night' },

  { key:'work',    color:'#7B5EA7', label:'Working',              emoji:'💼', note:'8 hrs/day, 5 days' },

  { key:'phone',   color:'#FF6B35', label:'Life wasted on screens', emoji:'📱', note:null, isPhone:true },

  { key:'eating',  color:'#E9A84C', label:'Eating & meals',       emoji:'🍽', note:'1.5 hrs/day' },

  { key:'chores',  color:'#A07855', label:'Chores & life admin',  emoji:'🧹', note:'1 hr/day' },

  { key:'free',    color:'#FFFFFF', label:'Free time',            emoji:'✨', note:null },

];

// ── RENDER RESULTS ───────────────────────────

function renderResults(name, age, phoneH) {

  const totalWeeks = LIFE_EXP * COLS;

  const livedWeeks = Math.round(Math.min(age * COLS, totalWeeks));

  const remWeeks   = totalWeeks - livedWeeks;


  const sleepFrac  = (SLEEP_H  * 7) / HRS_WEEK;

  const workFrac   = (WORK_H   * 5) / HRS_WEEK;

  const phoneFrac  = (phoneH   * 7) / HRS_WEEK;

  const eatFrac    = (EAT_H    * 7) / HRS_WEEK;

  const choresFrac = (CHORES_H * 7) / HRS_WEEK;

  const usedFrac   = sleepFrac + workFrac + phoneFrac + eatFrac + choresFrac;

  const freeFrac   = Math.max(0, 1 - usedFrac);


  const fracs = { sleep:sleepFrac, work:workFrac, phone:phoneFrac, eating:eatFrac, chores:choresFrac, free:freeFrac };


  const weekCounts = {};

  let assigned = 0;

  CATEGORIES.forEach((cat, i) => {

    if (i < CATEGORIES.length - 1) {

      weekCounts[cat.key] = Math.round(remWeeks * fracs[cat.key]);

      assigned += weekCounts[cat.key];

    } else {

      weekCounts[cat.key] = Math.max(0, remWeeks - assigned);

    }

  });


  const phoneWeeks = weekCounts['phone'];

  const yearsLost  = (phoneWeeks / 52).toFixed(1);

  const freeWeeks  = weekCounts['free'];


  // ── PERSONALISED HEADER ──

  document.getElementById('greeting').textContent = `${name}'s life in weeks`;

  document.getElementById('yearsLostEl').textContent = yearsLost;

  document.getElementById('resultsSubtext').textContent =

    `Based on ${phoneH} hours of screen time a day, you'll spend ${phoneWeeks.toLocaleString()} of your remaining weeks — that's ${yearsLost} years — staring at a screen.`;

  document.getElementById('ctaYears').textContent = yearsLost + ' years';


  // ── STATS ──

  document.getElementById('statsRow').innerHTML = `

    <div class="stat">

      <span class="stat-n" style="color:var(--yellow)">${totalWeeks.toLocaleString()}</span>

      <div class="stat-l">Total weeks<br/>in your life</div>

    </div>

    <div class="stat">

      <span class="stat-n" style="color:#444">${livedWeeks.toLocaleString()}</span>

      <div class="stat-l">Weeks already<br/>lived</div>

    </div>

    <div class="stat">

      <span class="stat-n" style="color:var(--orange)">${(remWeeks).toLocaleString()}</span>

      <div class="stat-l">Weeks<br/>remaining</div>

    </div>

    <div class="stat">

      <span class="stat-n" style="color:var(--orange)">${phoneWeeks.toLocaleString()}</span>

      <div class="stat-l">Weeks lost<br/>to screens</div>

    </div>

    <div class="stat">

      <span class="stat-n" style="color:var(--orange)">${yearsLost}y</span>

      <div class="stat-l">Years of life<br/>lost to screens</div>

    </div>

    <div class="stat">

      <span class="stat-n" style="color:rgba(255,255,255,0.85)">${freeWeeks.toLocaleString()}</span>

      <div class="stat-l">Genuinely free<br/>weeks remaining</div>

    </div>

  `;


  // ── BREAKDOWN BARS ──

  const bg = document.getElementById('breakdownGrid');

  bg.innerHTML = '';

  CATEGORIES.forEach(cat => {

    const wks = weekCounts[cat.key];

    const pct = remWeeks > 0 ? (wks / remWeeks * 100) : 0;

    const yrs = (wks / 52).toFixed(1);

    const row = document.createElement('div');

    row.className = 'brow';

    row.innerHTML = `

      <div class="brow-emoji">${cat.emoji}</div>

      <div class="brow-label">${cat.label}</div>

      <div class="brow-bar-wrap"><div class="brow-bar" style="width:${pct}%;background:${cat.color}"></div></div>

      <div class="brow-val" style="color:${cat.color}">${yrs}y

        <span style="font-family:'DM Mono',monospace;font-size:0.58rem;color:var(--muted)">(${wks.toLocaleString()}wk)</span>

      </div>

    `;

    bg.appendChild(row);

  });


  // ── DOT GRID ──

  const dots = [];

  for (let i = 0; i < livedWeeks; i++) dots.push('lived');

  CATEGORIES.forEach(cat => {

    for (let i = 0; i < weekCounts[cat.key]; i++) dots.push(cat.key);

  });

  while (dots.length < TOTAL_ROWS * COLS) dots.push('dead');


  const colorMap = { lived:'#1e1e1e', dead:'#0f0f0f' };

  CATEGORIES.forEach(c => { colorMap[c.key] = c.color; });


  const grid = document.getElementById('dotGrid');

  grid.innerHTML = '';

  for (let r = 0; r < TOTAL_ROWS; r++) {

    const row = document.createElement('div');

    row.className = 'dot-row';

    for (let c = 0; c < COLS; c++) {

      const d = document.createElement('div');

      const state = dots[r * COLS + c];

      d.className = 'dot';

      d.style.background = colorMap[state] || '#111';

      if (state === 'phone') d.style.boxShadow = '0 0 2px rgba(255,107,53,0.55)';

      row.appendChild(d);

    }

    grid.appendChild(row);

  }


  document.getElementById('gridLabel').textContent =

    `${LIFE_EXP}-year life expectancy (UK avg) · ${totalWeeks.toLocaleString()} weeks · 1 dot = 1 week`;


  // ── SIDE LABELS ──

  const ROW_H = 9;

  const livedRows = Math.ceil(livedWeeks / COLS);

  const blocks = [

    { color:'#1e1e1e', label:'Weeks already lived', note:null, rows:livedRows, isPhone:false, key:null, emoji:'' }

  ];

  CATEGORIES.forEach(cat => {

    blocks.push({

      color: cat.color, label: cat.label, note: cat.note,

      rows: Math.ceil(weekCounts[cat.key] / COLS),

      isPhone: !!cat.isPhone, key: cat.key, emoji: cat.emoji

    });

  });


  const sideDiv = document.getElementById('sideLabels');

  sideDiv.innerHTML = '';

  blocks.forEach(block => {

    const heightPx = Math.max(block.rows * ROW_H, 18);

    const el = document.createElement('div');

    el.className = 'side-label-block' + (block.isPhone ? ' phone-label' : '');

    el.style.height = heightPx + 'px';

    el.style.alignItems = 'center';

    const wks = block.key ? weekCounts[block.key] : livedWeeks;

    const yrs = (wks / 52).toFixed(1);

    const showLabel = heightPx >= 14;

    el.innerHTML = `

      <div class="side-dot" style="background:${block.color};${block.color==='#FFFFFF'?'border:1px solid rgba(255,255,255,0.4)':''}"></div>

      <div class="side-text">

        ${showLabel ? `<strong>${block.emoji ? block.emoji+' ' : ''}${block.label}</strong>` : ''}

        <em>${showLabel ? `${yrs}y (${wks.toLocaleString()}wk)${block.note ? ' · '+block.note : ''}` : ''}</em>

      </div>

    `;

    sideDiv.appendChild(el);

  });


  // ── LEGEND ──

  document.getElementById('legendDiv').innerHTML = `

    <div class="legend-item"><div class="ldot" style="background:#1e1e1e;border:1px solid #444"></div>Lived</div>

    ${CATEGORIES.map(c => `

      <div class="legend-item">

        <div class="ldot" style="background:${c.color};${c.color==='#FFFFFF'?'border:1px solid rgba(255,255,255,0.35)':''}"></div>

        ${c.label}

      </div>`).join('')}

  `;


}
